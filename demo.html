    onChange: function(data) {
        console.log('تم الاختيار:', data);
    }
});

خيارات متقدمة

const widget = new SaudiRegionsWidget({
    language: 'en',                    // ar أو en
    regionLabel: 'Region',             // تسمية حقل المنطقة
    cityLabel: 'City',                 // تسمية حقل المدينة
    regionPlaceholder: 'Select Region', // نص المنطقة الافتراضي
    cityPlaceholder: 'Select City',     // نص المدينة الافتراضي
    showLabels: true,                   // إظهار التسميات
    customStyles: `
        .saudi-regions-widget select {
            border: 2px solid #007cba;
            border-radius: 8px;
        }
    `,
    onChange: function(data) {
        if (data.region) {
            console.log('المنطقة:', data.region.name);
        }
        if (data.city) {
            console.log('المدينة:', data.city.name);
        }
    }
});

widget.render('container-id');

دوال مساعدة

// البحث في المناطق
const regions = widget.searchRegions('الرياض');

// البحث في المدن
const cities = widget.searchCities('جدة');

// الحصول على جميع المناطق
const allRegions = widget.getAllRegions();

// الحصول على منطقة بالمعرف
const region = widget.getRegionById('RD');

// الحصول على مدن منطقة معينة
const cities = widget.getCitiesByRegion('MQ');

// تعيين قيم
widget.setValue('RD', 'RD001'); // الرياض - الرياض

// الحصول على القيم الحالية
const currentValue = widget.getValue();

// إعادة تعيين
widget.reset();
